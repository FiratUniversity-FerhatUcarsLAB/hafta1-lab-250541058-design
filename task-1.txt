FONKSIYON CALISMA_PLANI_OLUSTUR (SINAV_TARIHI, KONULAR_LISTESI, TAHMINI_SURELER, GUNLUK_CALISMA_SAATI)

    // 1. HAZIRLIK ve DEĞERLENDİRME
    TOPLAM_GEREKEN_SURE = TOPLA (TAHMINI_SURELER)
    KALAN_GUN_SAYISI = HESAPLA_GUN_FARKI (BUGUN, SINAV_TARIHI)
    
    // Güvenlik Marjı ve Tekrar Süresi Ekleme
    TEKRAR_SURESI_YUZDESI = 0.20 // Toplam sürenin %20'si tekrar için
    TEKRAR_SURESI = TOPLAM_GEREKEN_SURE * TEKRAR_SURESI_YUZDESI
    TOPLAM_PLANLANACAK_SURE = TOPLAM_GEREKEN_SURE + TEKRAR_SURESI

    // 2. KONTROL: Planlama Mümkün mü?
    MAKSIMUM_CALISMA_SURESI = KALAN_GUN_SAYISI * GUNLUK_CALISMA_SAATI
    
    EĞER TOPLAM_PLANLANACAK_SURE > MAKSIMUM_CALISMA_SURESI İSE
        YAZDIR ("UYARI: Kalan süre, planlanan tüm konuları bitirmek için yetersiz. Günlük süreyi artırın veya konu kapsamını azaltın.")
        DUR
    SON_EĞER

    // 3. ÇALIŞMA PROGRAMINI OLUŞTURMA
    CALISMA_PLANI = BOS_LISTE()
    GECICI_TARIH = BUGUN
    KONU_SIRASI = SIRALA (KONULAR_LISTESI, ONCELIK_DUZEYINE_GORE) // Zor veya acil konular öne alınır

    HER BIR GUN (i=1'den KALAN_GUN_SAYISI'na) İÇİN TEKRAR ET
        GUNUN_PLANI = YENI_GUNLUK_PLAN()
        KALAN_CALISMA_HAKKI = GUNLUK_CALISMA_SAATI

        HER BIR KONU (K) İÇİN KONU_SIRASI'nda TEKRAR ET
            EĞER K.DURUMU = 'BITMEDI' VE KALAN_CALISMA_HAKKI > 0 İSE
                // Günlük çalışma süresi ve konunun kalan süresi karşılaştırılır
                PLANLANAN_SURE = MIN (K.KALAN_SURE, KALAN_CALISMA_HAKKI)
                
                EKLE_PLANA (GUNUN_PLANI, K.ADI, PLANLANAN_SURE)
                K.KALAN_SURE = K.KALAN_SURE - PLANLANAN_SURE
                KALAN_CALISMA_HAKKI = KALAN_CALISMA_HAKKI - PLANLANAN_SURE

                EĞER K.KALAN_SURE <= 0 İSE
                    K.DURUMU = 'TAMAMLANDI'
                    EKLE_PLANA (GUNUN_PLANI, K.ADI, "TEKRAR ET") // Konu bitince tekrar zamanı eklenir
                SON_EĞER
            SON_EĞER
        SON_TEKRAR

        EKLE_CALISMA_PLANI (CALISMA_PLANI, GECICI_TARIH, GUNUN_PLANI)
        GECICI_TARIH = GECICI_TARIH + 1_GUN

        // Son haftada sadece tekrar ve deneme sınavı planla
        EĞER GECICI_TARIH > SINAV_TARIHI - 7_GUN İSE
            EKLE_PLANA (GUNUN_PLANI, "GENEL_TEKRAR", GUNLUK_CALISMA_SAATI * 0.5)
            EKLE_PLANA (GUNUN_PLANI, "DENEME_SINAVI", GUNLUK_CALISMA_SAATI * 0.5)
        SON_EĞER

    SON_TEKRAR

    GERI_DÖN CALISMA_PLANI
SON_FONKSIYON
